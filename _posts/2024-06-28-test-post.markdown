```cpp
#include <stdio.h>

int main()
{
	int MaxLen = 10;
	printf("Inverted Triangle\n");
	for (int i = MaxLen; i > 0; --i)
	{
		for (int j = 0; j < i; ++j)
		{
			printf("*");
		}
		printf("\n");
	}

	printf("Pyramid\n");
	for (int i = MaxLen; i > 0; --i)
	{
		for (int j = 0; j < (i-1); ++j)
		{
			printf(" ");
		}
		for (int k = 0; k < (MaxLen-i+1)*2-1; ++k)
		{
			printf("*");
		}
		printf("\n");
	}


	return 0;
}
```
